#include <Servo.h>
Servo myServo;

int const potPin = A0;
int potVal;
int angle =  0;

void setup() {
    myServo.attach(9); // latur vita a? servomotorinn sé tengdur pinna 9
    Serial.begin(9600);
}

void loop(){
  potVal = analogRead(potPin); // potenitometer value = pinninn sem meterinn er tengdur
  Serial.print("potVal: ");
  Serial.print(potVal);
  angle = map(potVal, 0, 1023, 0, 179); // breytir rangeinu úr 0-1023 í 0-179
  Serial.print(", angle: "); 
  Serial.println(angle);
  myServo.write(angle);
  delay(15);
}
